
\newdualentry{com}{CoM}{Center of Mass}{is the position around which a
  spacecraft will rotate}
             
\newdualentry{eclss}{ECLSS}{Environmental Control and Life Support
  Systems}{are the systems onboard a spacecraft that control the
  capsule's pressure, temperature, carbon dioxide, and more}
             
\newdualentry{eva}{EVA}{Extravehicular Activity}{is any activity
  performed by an astronaut while outside of a spacecraft, in the
  vacuum of space}
             
\newglossaryentry{dv}{name=\dv, description={is the difference between
    two velocities; used to measure the energy required for orbital
    maneuvers; \seeref{sec:dv}}}
